drop table submitter;

drop table user;

create table user( 
email varchar(50),
phoneNo varchar(20),
firstName varchar(50),
lastName varchar(50),
countryName varchar(50),
password varchar(50),
passwordGenerated varchar(1),
userType varchar(15)
)

drop table journal;

create table journal(
id int not null auto_increment,
journalName varchar(100),
journalIcon longblob,
journalIconFileName varchar(100),
journalDescription varchar(500),
journalLongDescription varchar(5000),
journalBannerImage longblob,
journalBannerImageFileName varchar(100),
primary key (id)
)


drop table submissions;

create table submissions(
email varchar(50),
journal varchar(50),
articleType varchar(50),
menuScriptTitle varchar(1000),
abstractName varchar(4000),
attachmentFileName varchar(100),
attachment longblob)

drop table editor;

create table editor(
id int not null auto_increment,
firstName varchar2(50),
lastName varchar2(50),
attachment longblob,
description varchar2(200),
affiliation varchar2(200),
journalId int);

drop table reviewer;
create table reviewer(
id int not null auto_increment,
firstName varchar2(50),
lastName varchar2(50),
email varchar2(200),
country varchar2(70));
